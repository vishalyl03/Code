#include<stdio.h>


// Problem: digit sum with function Declaration

int sum(int var1,int var2);     // Function Declaration

int main()
{
    int a,b;
    scanf("%d %d",&a,&b);
    
    int result=sum(a,b);   // Function call 
    printf("Sum of %d and %d is %d",a,b,result);
    
    return 0;
}
// Function Definition
int sum(int num1,int num2)    
{
    int val=num1+num2;
    return val;
}
